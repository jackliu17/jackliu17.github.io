加载plist中的数据

该方法并不用在viewDidLoad中调用，而是在实际取值时调用

- (NSArray *)imageDicts
{
//判断数组为空时建立
if (!_imageDicts) {
   //从NSBundle中读取路径
    NSString *path = [[NSBundle mainBundle] pathForResource:@"imageDate.plist" ofType:nil];
    //利用文件路创建数组
    _imageDicts = [NSArray arrayWithContentsOfFile:path];
}
//返回数组
return _imageDicts;
}
