发表于 2016-05-18   |   分类于 IT   |  

读取数组

let arr = NSArray(contentsOfURL: NSURL(fileURLWithPath: NSBundle.mainBundle().pathForResource("arr", ofType: "plist")!))
print(arr)

读取字典

let dict = NSDictionary(contentsOfURL: NSURL(fileURLWithPath: NSBundle.mainBundle().pathForResource("data", ofType: "plist")!))
print(dict)

let name:String = dict!["name"] as! String
print("name : \(name)")
